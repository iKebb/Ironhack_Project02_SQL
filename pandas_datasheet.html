3. Data Cleaning & Preprocessing

Missing Value Detection & Treatment
# Check for missing values print("Missing values per column:") print(df.isnull().sum()) # Visualize missing values plt.figure(figsize=(10, 6)) sns.heatmap(df.isnull(), cbar=True, yticklabels=False) plt.title('Missing Values Heatmap') plt.show() # Handle missing values df_cleaned = df.copy() # Drop columns with too many missing values df_cleaned = df_cleaned.dropna(axis=1, thresh=len(df_cleaned)*0.7) # Fill numerical missing values df_cleaned['income'] = df_cleaned['income'].fillna(df_cleaned['income'].median()) # Fill categorical missing values df_cleaned['country'] = df_cleaned['country'].fillna('Unknown') # Forward fill for time series df_cleaned = df_cleaned.fillna(method='ffill')
Data Type Conversion & Validation
# Convert to proper data types df['age'] = pd.to_numeric(df['age'], errors='coerce') df['signup_date'] = pd.to_datetime(df['signup_date'], errors='coerce') df['subscription_type'] = df['subscription_type'].astype('category') # Handle string columns with commas (like "1,000,000") df['income'] = pd.to_numeric(df['income'].astype(str).str.replace(',', ''), errors='coerce') # Check data types after conversion print("Data types after conversion:") print(df.dtypes)

Outlier Detection & Treatment
# Detect outliers using IQR method def detect_outliers_iqr(data, column): Q1 = data[column].quantile(0.25) Q3 = data[column].quantile(0.75) IQR = Q3 - Q1 lower_bound = Q1 - 1.5 * IQR upper_bound = Q3 + 1.5 * IQR outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)] return outliers # Find outliers in income income_outliers = detect_outliers_iqr(df, 'income') print(f"Found {len(income_outliers)} outliers in income") # Cap outliers (Winsorization) df['income_capped'] = np.clip(df['income'], df['income'].quantile(0.05), df['income'].quantile(0.95))

11. Best Practices & Performance

Performance Optimization
# Use efficient data types df_optimized = df.copy() # Convert to categorical for low-cardinality columns categorical_columns = ['country', 'subscription_type', 'engagement_level'] for col in categorical_columns: df_optimized[col] = df_optimized[col].astype('category') # Use downcasting for numerical columns df_optimized['age'] = pd.to_numeric(df_optimized['age'], downcast='integer') df_optimized['income'] = pd.to_numeric(df_optimized['income'], downcast='float') # Memory usage comparison original_memory = df.memory_usage(deep=True).sum() optimized_memory = df_optimized.memory_usage(deep=True).sum() print(f"Original memory: {original_memory / 1024:.1f} KB") print(f"Optimized memory: {optimized_memory / 1024:.1f} KB") print(f"Reduction: {((original_memory - optimized_memory) / original_memory * 100):.1f}%")
